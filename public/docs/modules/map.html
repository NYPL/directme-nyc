  function _init() {

    $('.borough').on('click', function(event) {
      $.when(loadDV.init(event.target.id)).done(function() {
        log("loaded!");
        $('html, body').animate({
          scrollTop: $(loadDV.loaded()).offset().top
        }, 200);
      });
      return false;
    });
  }
